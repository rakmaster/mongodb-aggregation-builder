Quick Example

<template>
  <MongoAggregationBuilder
    :connection="mongoConnection"
    :theme="customTheme"
  />
</template>

<script setup>
import { MongoAggregationBuilder } from 'mongo-aggregation-builder'

const mongoConnection = {
  uri: 'mongodb://localhost:27017',
  database: 'myapp',
  collection: 'users'
}

const customTheme = {
  'bg-color': '#1a1a1a',
  'text-color': '#ffffff'
}
</script>
